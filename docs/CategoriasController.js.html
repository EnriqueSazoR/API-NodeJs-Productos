import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();

/**
 * Crear una nueva categoría
 * 
 * @async
 * @function PostCategoria
 * @param {object} req - Objeto de solicitud (Express Request) 
 * @param {object} res - Objeto de respuesta (Express Response) 
 * @returns {json} Mensaje de confirmación o error
 * 
 * @example
 * // Body esperado:
 * {
 *    "nombre" : "Laptops"
 * }
 * 
 * // Response 200
 * {  
 *    "mensaje" : "categoría creada"
 * }
 * 
 * // Response 400
 * {
 *    "error" : "Categoría Laptops ya existe"
 * }
 */
export const PostCategoria = async (req, res) => {
  try {
    const { nombre } = req.body;
    
    const categoriaExistente = await prisma.Categoria.findFirst({
      where: { nombre },
    });

    if (categoriaExistente) {
      return res.status(400).json({ error: `Categoria ${nombre} ya existe` });
    }

    
    await prisma.Categoria.create({
      data: {
        nombre,
      },
    });
    res.json({ mensaje: "Categoría creada" });
  } catch (error) {
    console.log(`Error al registrar categoria ${error}`);
    res.status(500).json({ error: "Error en el servidor" });
  }
};

/**
 * Obtener todas las categorías
 * 
 * @async
 * @function GetCategorias
 * @param {object} req - Objeto de solicitud 
 * @param {object} res - Objeto de respuesta
 * @returns {json} Lista de categorías
 * 
 * @example
 * // Response 200
 * {
 *   "marcas" : [
 *      {"nombre" : "Laptops"},
 *      {"nombre" : "Video Juegos"}
 *   ]
 * }
 * 
 * // Response 500
 * {
 *    "error" : "Error en el servidor"
 * }
 */
export const GetCategorias = async (req, res) => {
  try {
    const listaCategorias = await prisma.Categoria.findMany({
      select: {
        nombre: true,
      },
    });
    res.send({ categorias: listaCategorias });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en el servidor" });
  }
};

/**
 * Actualizar una categoría
 * 
 * @async
 * @function PutCategoria
 * @param {object} req - Objeto de solicitud Express
 * @param {object} req.params - Parámetros de la URL
 * @param {number} req.params.id  - ID de la categoría a actualizar
 * @param {string} req.body.nombre - Nuevo nombre de la categoría
 * @param {object} res - Objeto de respuesta Express
 * @returns {json} Mensaje de éxito o error
 * 
 * @example
 * // Request URL
 * PUT categorias/actualizar/1
 * 
 * // Request Body
 * {
 *    "nombre" : "Laptops"
 * }
 * 
 * // Response 200
 * { 
 *     "mensaje" : "Marca Actualizada",
 *     "marca" : {
 *        "nombre" : "Laptops",
 *        "updatedAt" : "2025-09-10T22:00:00.000Z"
 *   }
 * }
 * 
 * // Response 400
 * {
 *    "error" : "Categoría no encontrada"
 * }
 * 
 * // Response 500
 * {  
 *    "error" : "Error en servidor"
 * }
 */
export const PutCategoria = async (req, res) => {
  try {
    const { id } = req.params;
    const { nombre } = req.body;

    const categoria = await prisma.Categoria.findUnique({
      where: { id: parseInt(id) },
    });

    if (!categoria) {
      return res.status(400).json({ error: "Categoría no encontrada" });
    }

    
    const categoriaExistente = await prisma.Categoria.findFirst({
      where: { nombre },
    });
    if (categoriaExistente) {
      return res
        .status(400)
        .json({ error: `Categoría ${nombre} ya existe` });
    }

    
    const categoriaActualizada = await prisma.Categoria.update({
      where: { id: parseInt(id) },
      data: {
        nombre: nombre ?? categoria.nombre,
      },
      select: {
        nombre: true,
        updatedAt: true,
      },
    });
    res.json({
      exito: "Categoría Actualizada",
      categoria: categoriaActualizada,
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en servidor" });
  }
};
