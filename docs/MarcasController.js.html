// Controlador para CRUD de marcas
import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();


/**
 * Crear una nueva marca
 * 
 * @async
 * @function PostMarca
 * @param {object} req - Objeto de solicitud (Express Request) 
 * @param {object} res - Objeto de respuesta (Express Response) 
 * @returns {json} Mensaje de confirmación o error
 * 
 * @example
 * // Body esperado:
 * {
 *    "nombre" : "Nike"
 * }
 * 
 * // Response 200
 * {  
 *    "mensaje" : "marca creada"
 * }
 * 
 * // Response 400
 * {
 *    "error" : "Marca Nike ya existe"
 * }
 */
export const PostMarca = async (req, res) => {
  try {

    const { nombre } = req.body;
    
    const marcaExistente = await prisma.Marca.findFirst({
      where: { nombre },
    });

    if (marcaExistente) {
      return res.status(400).json({ error: `Marca ${nombre} ya existe` });
    }

    await prisma.Marca.create({
      data: {
        nombre,
      },
    });
    res.json({ mensaje: "Marca creada" });
  } catch (error) {
    console.error(`Error al crear marca ${error}`);
    res.status(500).json({ error: "Error en el servidor" });
  }
};


/**
 * Obtener todas las marcas
 * 
 * @async
 * @function GetMarcas
 * @param {object} req - Objeto de solicitud 
 * @param {object} res - Objeto de respuesta
 * @returns {json} Lista de marcas 
 * 
 * @example
 * // Response 200
 * {
 *   "marcas" : [
 *      {"nombre" : "Nike"},
 *      {"nombre" : "Adidas"}
 *   ]
 * }
 * 
 * // Response 500
 * {
 *    "error" : "Error en el servidor"
 * }
 */
export const GetMarcas = async (req, res) => {
  try {
    const listaMarcas = await prisma.Marca.findMany({
      select: {
        nombre: true,
      },
    });
    res.send({ marcas: listaMarcas });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en el servidor" });
  }
};


/**
 * Actualizar una marca
 * 
 * @async
 * @function PutMarca
 * @param {object} req - Objeto de solicitud Express
 * @param {object} req.params - Parámetros de la URL
 * @param {number} req.params.id  - ID de la marca a actualizar
 * @param {string} req.body.nombre - Nuevo nombre de la marca
 * @param {object} res - Objeto de respuesta Express
 * @returns {json} Mensaje de éxito o error
 * 
 * @example
 * // Request URL
 * PUT marcas/actualizar/1
 * 
 * // Request Body
 * {
 *    "nombre" : "Puma"
 * }
 * 
 * // Response 200
 * { 
 *     "mensaje" : "Marca Actualizada",
 *     "marca" : {
 *        "nombre" : "Puma",
 *        "updatedAt" : "2025-09-10T22:00:00.000Z"
 *   }
 * }
 * 
 * // Response 400
 * {
 *    "error" : "Marca no encontrada"
 * }
 * 
 * // Response 500
 * {  
 *    "error" : "Error en el servidor"
 * }
 */
export const PutMarca = async (req, res) => {
  try {
    const { id } = req.params;
    const { nombre } = req.body;
    const marca = await prisma.Marca.findUnique({
      where: { id: parseInt(id) },
    });

    if (!marca) {
      return res.status(400).json({ error: "Marca no encontrada" });
    }

    const marcaExistente = await prisma.Marca.findFirst({
      where: { nombre },
    });

    if (marcaExistente) {
      return res.status(400).json({ error: `La marca ${nombre} ya existe` });
    }

    const marcaActualizada = await prisma.Marca.update({
      where: { id: parseInt(id) },
      data: {
        nombre: nombre ?? marca.nombre,
      },
      select: {
        nombre: true,
        updatedAt: true,
      },
    });
    res.json({
      mensaje: "Marca actualizada",
      marca: marcaActualizada,
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en el servidor" });
  }
};
