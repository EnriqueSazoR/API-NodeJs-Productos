import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();

/**
 * Crear un producto nuevo
 * 
 * @async
 * @function PostProducto
 * @param {object} req - Objeto de solicitud (Express Request) 
 * @param {object} res - Objeto de respuesta (Express Reponse)
 * @returns {json} Mensaje de confirmación o error
 * 
 * @example
 * // Body esperado;
 * {
 *    "nombre" : "Xbox Series X",
 *    "stock" : 2,
 *    "categoria" : "Consolas",
 *    "marca" : "Microsoft"
 * }
 * 
 * 
 * // Response 200
 * { 
 *    mensaje : "Producto creado" 
 * }
 * 
 * // Response 400
 * }
 *    error : "El producto ya existe"  
 * }
 * 
 * // Response 404
 * {
 *    error : "Categoría no encontrada" o "Marca no econtrada"
 * }
 */
export const PostProducto = async (req, res) => {
  try {
    const { nombre, stock, categoria, marca } = req.body;

    const productoExistente = await prisma.Producto.findFirst({
      where : {nombre}
    })
    if(productoExistente) {
      return res.status(400).json({error : `EL producto ${nombre} ya existe`})
    }
    const categoriaObj = await prisma.Categoria.findFirst({
      where: { nombre: categoria.trim() },
    });
    if (!categoriaObj) {
      return res.status(404).json({ error: "Categoría no encontrada" });
    }
    const marcaObj = await prisma.Marca.findFirst({
      where: { nombre: marca.trim() },
    });
    if (!marcaObj) {
      return res.status(404).json({ error: "Marca no encontrada" });
    }

    const producto = await prisma.Producto.create({
      data: {
        nombre,
        stock,
        categoria: {
          connect: { id: categoriaObj.id },
        },
        marca: {
          connect: { id: marcaObj.id },
        },
      },
    });
    res.status(200).json({ mensaje: "Producto creado" });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en servidor" });
  }
};

/**
 * Obtener todos los productos
 * 
 * @async
 * @function GetProdutcos
 * @param {object} req - Objeto de solicitud 
 * @param {object} res - Objeto de respuesta
 * @returns {json} Lista de productos
 * 
 * @example
 * // Response 200
 * {nombre:"nnn","categoria:"nnn",marca:"nnn"},
 * {nombre:"nnn1","categoria:"nnn1",marca:"nnn1"}
 * 
 * // Response 500
 * {
 *    error : "Error en el servidor"
 * }
 * 
 */
export const GetProdutcos = async (req, res) => {
  try {
    const listaProductos = await prisma.Producto.findMany({
      select: {
        nombre: true,
        categoria: {
          select: {
            nombre: true,
          },
        },
        marca: {
          select: {
            nombre: true,
          },
        },
      },
    });
    res.send({
      Productos: listaProductos,
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en el servidor" });
  }
};


/**
 * Actualizar un producto
 * 
 * @async
 * @function PutProducto
 * @param {object} req.body - Objeto de solicitud Express
 * @param {object} req.params - Parámetros de la URL
 * @param {number} req.params.id  - ID del producto a actualizar
 * @param {object} res - Objeto de respuesta Express
 * @returns {json} Mensaje de confirmación o error
 * 
 * @example
 * 
 * // Response 200
 * {
 *    {exito : "Producto Actualizado"},
 * 
 *    "nombre" : "nnn",
 *    "stock" : 00,
 *    "categoria" : "nnn",
 *    "marca" : "nnn",
 *    "fechaActualizacion" : "yyyy/mm/dd"
 * }
 * 
 * // 
 */
export const PutProducto = async (req, res) => {
  try {
    const { id } = req.params;
    const { nombre, stock, categoria, marca } = req.body;

    const buscarProducto = await prisma.Producto.findUnique({
      where: { id: parseInt(id) },
    });

    if (!buscarProducto) {
      res.status(404).json({ error: "Producto no encontrado" });
    }

    const categoriaObj = await prisma.Categoria.findFirst({
      where : {nombre : categoria.trim()}
    })
    if(!categoriaObj) {
      return res.status(404).json({error : "La categoria no existe"})
    }

    const marcaObj = await prisma.Marca.findFirst({
      where : {nombre : marca.trim()}
    })
    if(!marcaObj) {
      return res.status(404).json({error : "La marca no existe"})
    }

    const productoActualizado = await prisma.Producto.update({
      where: { id: parseInt(id) },
      data: {
        nombre: nombre ?? producto.nombre,
        stock: stock ?? producto.stock,
        categoria : {
          connect : {id : categoriaObj.id}
        },
        marca : {
          connect : {id : marcaObj.id}
        }
      },
      select: {
        nombre: true,
        stock: true,
        categoria : {
          select : {
            nombre : true
          }
        },
        marca : {
          select : {
            nombre : true
          }
        },
        fechaActualizacion: true,
      },
    });
    res.json({
      exito: "Producto Actualizado",
      producto: productoActualizado,
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error en servidor" });
  }
};
