import { PrismaClient } from "@prisma/client";
import bcrypt from "bcrypt";
const prisma = new PrismaClient();

/**
 * Buscar un producto por nombre
 * 
 * @async
 * @function buscarProducto
 * @param {object} req - Objeto de solicitud tomando como parametro {nombre}
 * @param {object} res - Objeto de respuesta  
 * @returns {json} Mensaje de confirmación o error
 * - 200 'OK' --> Coincidencia del producto encontrado
 * - 400 'Bad Request' --> Si no se encuentra el producto
 * - 500 'Internal Server Error' --> Si hay un error en el servidor
 * 
 * @example
 * {
 *    "nombre" : "Samsung Galaxy Tab",
 *    "stock" : 1
 * }
 */
export const buscarProducto = async (req, res) => {
  try {
    const { nombre } = req.body;
    const productoBD = await prisma.Producto.findFirst({
      where: {
        nombre: {
          equals: nombre,
        },
      },
      select: {
        nombre: true,
        stock: true,
      },
    });
    if (!productoBD) {
      return res
        .status(400)
        .json({ error: "No se encontró el producto buscado" });
    }

    res.status(200).json({ productoBD });
  } catch (error) {
    console.error(error);
    return res.status(500).json({ error: "Error en servidor" });
  }
};

/**
 * Busca productos por marca
 * 
 * @async
 * @function buscarProductoPorMarca
 * @param {object} req - Objeto de solicitud tomando como parametro la marca 
 * @param {object} res - Objeto de respuesta 
 * @returns {json} Mensaje de confirmación o error
 * - 200 'OK' --> Lista de productos
 * - 500 'Internal Server Error' --> Si hay problemas en el servidor
 * 
 * @example
 *    {respuesta : "Productos relacionados con {marca}"},
 *      {
 *          "nombre" : "nnn",
 *          "stock" : 00
 *      },
 *      {
 *          "nombre" : "nnn1",
 *          "stock" : 11
 *      },
 * 
 */
export const buscarProductoPorMarca = async (req, res) => {
  try {
    const { nombre } = req.body;
    const marcaBD = await prisma.Marca.findMany({
      where: { nombre },
      include: {
        productos: {
          select: {
            nombre: true,
            stock: true,
          },
        },
      },
    });

    const productos = marcaBD.flatMap((m) => m.productos);

    res.status(200).json({
      respuesta: `Productos relaciondos con ${nombre}`,
      productos,
    });
  } catch (error) {
    console.error(error);
    return res.status(500).json({ error: "Error en servidor" });
  }
};

/**
 * Actualizar credenciales del usuario
 * 
 * @async
 * @function actualizarCredenciales
 * @param {object} req - Objeto de solicitud tomando como parametros {usuario, password}
 * un middleware valida que los datos ingresados cumplan con los estandares requeridos
 * @param {object} res - Objeto de respuesta 
 * @returns {json} Mensaje de confirmación o error
 * - 200 'OK' --> Si la actualización es correcta
 * - 400 'Bad Request' --> Si el usuario es incorrecto o se intenta ingresar un duplicado
 * - 500 'Internal Server Error' --> Si hay un problema con el servidor
 * 
 * @example
 * {
 *    "usuario" : "lsazoo":
 *    "password" : "WebDev87#"
 * }
 * 
 */
export const actualizarCredenciales = async (req, res) => {
  try {
    const { id } = req.params;
    const { usuario, password } = req.body;

    const buscarUsuario = await prisma.Usuario.findUnique({
      where: { id: parseInt(id) },
    });
    if (!buscarUsuario) {
      return res.status(400).json({ error: "Usuario Incorrecto" });
    }

    const usuarioDuplicado = await prisma.Usuario.findUnique({
      where: { usuario },
    });
    if (usuarioDuplicado) {
      return res
        .status(400)
        .json({ error: "El nombre de usuario ingresado ya existe" });
    }

    // Hashear la nueva contraseña
    const nuevaContraseña = await bcrypt.hash(password, 10);

    const credencialesActualizadas = await prisma.Usuario.update({
      where: { id: parseInt(id) },
      data: {
        usuario: usuario ?? buscarUsuario.usuario,
        password: nuevaContraseña,
      },
      select: {
        usuario: true,
      },
    });

    res.status(200).json({
      completo: "Actualización de usuario correcta",
      credencialesActualizadas,
    });
  } catch (error) {
    console.error(error);
    return res.status(500).json({ error: "Error en el servidor" });
  }
};
